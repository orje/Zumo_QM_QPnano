title lcd sm


[*] -down-> start

state start {
state start : lcd.print(F("Press Start"));
}
start -down-> clear : BUTTON_SIG

state clear {
state clear : lcd.clear();
}
clear -down-> stop : TRAPPED_SIG

state stop {
state stop : lcd.print(F("Press Stopp"));
}
stop -down-> start : BUTTON_SIG
